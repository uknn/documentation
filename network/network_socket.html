inherit

		undefine

feature 

	exists: BOOLEAN
			-- Does socket exist?
		do
			Result := descriptor_available and then (fd > 0 or else fd1 > 0)
		ensure then
			definition: Result implies descriptor_available
		end

	close_socket
		do
			if not is_closed then
				if is_created then
					c_close (fd, fd1)
				end
				is_closed := True
			end
			descriptor_available := False
			is_open_read := False
			is_open_write := False
		end

	descriptor: INTEGER_32
			-- Socket descriptor of current socket
		do
			Result := fd
		end

	connect
			-- Connect socket to peer address.
		local
			retried: BOOLEAN
		do
			if not retried then
				do_connect (peer_address)
				is_connected := True
				is_open_write := True
				is_open_read := True
			end
		rescue
			if not assertion_violation then
				is_connected := False
				is_open_read := False
				is_open_write := False
				retried := True
				retry
			end
		end

	bind
			-- Bind socket to local address in address.
		local
			retried: BOOLEAN
		do
			if not retried then
				do_bind (address)
				is_open_read := True
				is_bound := True
			end
		rescue
			if not assertion_violation then
				is_bound := False
				is_open_read := False
				retried := True
				retry
			end
		end
	
feature -- Status report

	address_type: NETWORK_SOCKET_ADDRESS
			-- Type of address and peer_address
		do
			check
				False
			then
			end
		end

	is_closed: BOOLEAN
			-- Is socket closed?

	is_created: BOOLEAN

	is_connected: BOOLEAN

	is_bound: BOOLEAN

	port: INTEGER_32
			-- Port socket is bound to.
		do
			if not is_bound then
				Result := -1
			else
				Result := internal_port
			end
		end

	reuse_address: BOOLEAN
			-- Is reuse_address option set?
		require
			socket_exists: exists
		local
			reuse: INTEGER_32
		do
			reuse := c_get_sock_opt_int (descriptor, level_sol_socket, so_reuse_addr)
			Result := reuse /= 0
		end

	is_valid_peer_address (addr: like address): BOOLEAN
			-- Is `addr' a valid peer address?
		do
			Result := (addr.family = af_inet or else addr.family = af_inet6)
		end

	is_valid_family (addr: like address): BOOLEAN
			-- Is `addr' the same family as Current?
		do
			Result := (addr.family = af_inet or else addr.family = af_inet6)
		end

	ready_for_reading: BOOLEAN
			-- Is data available for reading from the socket within
			-- timeout seconds?
		require
			socket_exists: exists
		local
			retval: INTEGER_32
		do
			retval := c_select_poll_with_timeout (descriptor, True, timeout)
			Result := (retval > 0)
		end

	ready_for_writing: BOOLEAN
			-- Can data be written to the socket within timeout seconds?
		require
			socket_exists: exists
		local
			retval: INTEGER_32
		do
			retval := c_select_poll_with_timeout (descriptor, False, timeout)
			Result := (retval > 0)
		end

	has_exception_state: BOOLEAN
			-- Is socket in exception state within timeout seconds?
		require
			socket_exists: exists
		local
			retval: INTEGER_32
		do
			retval := c_check_exception_with_timeout (descriptor, timeout)
			Result := (retval > 0)
		end

	timeout: INTEGER_32
			-- Duration of timeout in seconds
	
feature -- Status setting

	set_reuse_address
			-- Turn reuse_address option on.
		require
			socket_exists: exists
		do
			c_set_sock_opt_int (descriptor, level_sol_socket, so_reuse_addr, 1)
		end

	do_not_reuse_address
			-- Turn reuse_address option off.
		require
			socket_exists: exists
		do
			c_set_sock_opt_int (descriptor, level_sol_socket, so_reuse_addr, 0)
		end

	set_timeout (n: INTEGER_32)
			-- Set timeout to `n' seconds.
		require
			non_negative: n >= 0
		do
			if n > 0 then
				timeout := n
			else
				timeout := Default_timeout
			end
		ensure
			timeout_set: timeout = n or timeout = Default_timeout
		end

	set_non_blocking
			-- Set socket to non-blocking mode.
		do
			c_set_non_blocking (fd)
			if fd1 > 0 then
				c_set_non_blocking (fd1)
			end
			is_blocking := False
		end

	set_blocking
			-- Set socket to blocking mode.
		do
			c_set_blocking (fd)
			if fd1 > 0 then
				c_set_blocking (fd1)
			end
			is_blocking := True
		end
	
feature {NONE} -- Implementation

	make_socket
			-- Create socket descriptor.
		do
			do_create
			if fd > -1 then
				is_closed := False
				descriptor_available := True
				set_blocking
			end
		end

	shutdown
		do
			c_shutdown (fd, fd1)
		end

	fd: INTEGER_32

	fd1: INTEGER_32

	last_fd: INTEGER_32

	internal_port: INTEGER_32
			-- Internal storage for port when is_bound.

	do_create
		deferred
		end

	do_connect (a_peer_address: like peer_address)
		require
			a_peer_address_attached: a_peer_address /= Void
		deferred
		end

	do_bind (a_address: like address)
		require
			a_address_attached: a_address /= Void
		deferred
		end
	
feature {NONE} -- Constants

	Default_timeout: INTEGER_32 = 20
			-- Default timeout duration in seconds

			"C blocking"

		external
			"C blocking"
		end

	c_shutdown (an_fd: INTEGER_32; an_fd1: INTEGER_32)
		external
			"C blocking"
		alias
			"en_socket_shutdown"
		end

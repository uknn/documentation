note
	description: "Encoding of arbitrary objects graphs between sessions of a same program."
	legal: "See notice at end of class."
	status: "See notice at end of class."
	date: "$Date: 2010-02-24 14:17:13 -0800 (Wed, 24 Feb 2010) $"
	revision: "$Revision: 82421 $"

class 
	SED_BASIC_SERIALIZER

inherit
	SED_SESSION_SERIALIZER
		redefine
			write_header,
			setup_version
		end

create 
	make

feature {NONE} -- Implementation

	setup_version
			-- Set version with the appropriate version number.
		do
			version := {SED_VERSIONS}.basic_version_6_6
		end

	write_header (a_list: ARRAYED_LIST [ANY])
			-- Write header of storable.
		local
			l_dtype_table: like type_table
			l_ser: like serializer
			l_int: like internal
			l_dtype: INTEGER_32
		do
			l_ser := serializer
			l_ser.write_compressed_natural_32 (version)
			from
				l_int := internal
				l_dtype_table := type_table (a_list)
				l_ser.write_compressed_natural_32 (l_dtype_table.count.to_natural_32)
				l_dtype_table.start
			until
				l_dtype_table.after
			loop
				l_dtype := l_dtype_table.item_for_iteration
				l_ser.write_compressed_natural_32 (l_dtype.to_natural_32)
				l_ser.write_string_8 (l_int.type_name_of_type (l_dtype))
				l_dtype_table.forth
			end
			write_object_table (a_list)
		end

	type_table (a_list: ARRAYED_LIST [ANY]): HASH_TABLE [INTEGER_32, INTEGER_32]
			-- Given a list of objects `a_list', builds a compact table of the
			-- dynamic type IDs present in `a_list'.
		require
			a_list_not_void: a_list /= Void
			a_list_not_empty: not a_list.is_empty
		local
			l_dtype: INTEGER_32
			l_int: like internal
			l_area: SPECIAL [ANY]
			i, nb: INTEGER_32
		do
			l_int := internal
			from
				create Result.make (500)
				l_area := a_list.area
				i := 0
				nb := a_list.count
			until
				i = nb
			loop
				l_dtype := l_int.dynamic_type (l_area.item (i))
				Result.put (l_dtype, l_dtype)
				i := i + 1
			end
		ensure
			type_table_not_void: Result /= Void
		end

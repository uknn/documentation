note
	description: "[
							Fait afficher à l'écran les divers sprites
							Affiche à l'écran les sprites qui lui sont envoyé par les classes approprié (megaman,background etc.) et contient aussi les coordonnées des divers sprites
		
	]"
	author: "Samuel Forcier - David Lafrenière"
	date: "25 mars 2013"
	revision: "1.0"

deferred class 
	SPRITE

inherit
	DISPOSABLE

feature {NONE} -- initialise les parametre par defaut d'un texte

	initialise_text (a_chemin: STRING_8)
		local
			l_chemin_c: C_STRING
			l_r: NATURAL_8
			l_g: NATURAL_8
			l_b: NATURAL_8
			l_texte_c: C_STRING
		do
			create l_chemin_c.make (a_chemin)
			text_size := 26
			l_r := 255.to_natural_8
			l_g := 255.to_natural_8
			l_b := 255.to_natural_8
			sdl_color := sdl_color.memory_calloc ({RAPPER}.sizeof_sdl_color, 1)
			{RAPPER}.set_r_from_sdl_color (sdl_color, l_r)
			{RAPPER}.set_g_from_sdl_color (sdl_color, l_g)
			{RAPPER}.set_b_from_sdl_color (sdl_color, l_b)
			ttf_font := {RAPPER}.ttf_openfont (l_chemin_c.item, text_size)
			create l_texte_c.make (texte)
			ptr := {RAPPER}.ttf_rendertext_blended (ttf_font, l_texte_c.item, sdl_color)
			x := 0.to_real
			y := 0.to_real
			x_rect := 0
			y_rect := 0
			w_rect := {RAPPER}.get_w_from_sdl_surface (ptr)
			h_rect := {RAPPER}.get_h_from_sdl_surface (ptr)
		end
	
feature {NONE} -- modifie le texte d'un object texte

	modifie_texte (a_texte: STRING_8)
		local
			l_texte_c: C_STRING
		do
			create l_texte_c.make (a_texte)
			ptr := {RAPPER}.ttf_rendertext_blended (ttf_font, l_texte_c.item, sdl_color)
			w_rect := {RAPPER}.get_w_from_sdl_surface (ptr)
			h_rect := {RAPPER}.get_h_from_sdl_surface (ptr)
		end
	
feature {NONE} -- Initialise les parametre par defaut d'un sprite

	initialize (a_chemin: STRING_8)
			-- Initialise les parametre par defaut d'un sprite	
		local
			l_chemin_c: C_STRING
		do
			create l_chemin_c.make (a_chemin)
			ptr := {RAPPER}.sdl_loadbmp (l_chemin_c.item)
			x := 0.to_real
			y := 0.to_real
			x_rect := 0
			y_rect := 0
			w_rect := {RAPPER}.get_w_from_sdl_surface (ptr)
			h_rect := {RAPPER}.get_h_from_sdl_surface (ptr)
		end
	
feature --Affiche un sprite

	sprite_affiche (a_screen: POINTER)
			--Affiche un sprite
		local
			l_screen, l_targetarea, l_null: POINTER
			l_inutile: INTEGER_32
			l_chemin_c: C_STRING
			l_rect_src: POINTER
		do
			l_screen := a_screen
			l_targetarea := l_targetarea.memory_calloc ({RAPPER}.sizeof_sdl_rect, 1)
			l_rect_src := l_targetarea.memory_calloc ({RAPPER}.sizeof_sdl_rect, 1)
			{RAPPER}.set_x_from_sdl_rect (l_targetarea, x.truncated_to_integer.as_integer_16)
			{RAPPER}.set_y_from_sdl_rect (l_targetarea, y.truncated_to_integer.as_integer_16)
			{RAPPER}.set_w_from_sdl_rect (l_targetarea, 0.to_natural_16)
			{RAPPER}.set_h_from_sdl_rect (l_targetarea, 0.to_natural_16)
			{RAPPER}.set_x_from_sdl_rect (l_rect_src, x_rect.as_integer_16)
			{RAPPER}.set_y_from_sdl_rect (l_rect_src, y_rect.as_integer_16)
			{RAPPER}.set_w_from_sdl_rect (l_rect_src, w_rect.as_natural_16)
			{RAPPER}.set_h_from_sdl_rect (l_rect_src, h_rect.as_natural_16)
			l_inutile := {RAPPER}.sdl_blitsurface (ptr, l_rect_src, l_screen, l_targetarea)
			l_targetarea.memory_free
		end

	set_transparency
			-- Defini la transparence d'un image
		local
			l_rgb: NATURAL_32
			l_format: POINTER
			l_inutile: INTEGER_32
		do
			l_format := {RAPPER}.get_format_from_sdl_surface (ptr)
			l_rgb := {RAPPER}.sdl_maprgb (l_format, 255.to_natural_8, 0.to_natural_8, 255.to_natural_8)
			l_inutile := {RAPPER}.sdl_setcolorkey (ptr, {RAPPER}.sdl_srccolorkey, l_rgb)
		end

	dispose
			-- Libère un image de la mémoire
		do
			{RAPPER}.sdl_freesurface (ptr)
		end

	set_x_rect (nouveaux: INTEGER_32)
			-- assignateur de la variable x_rect
		do
			x_rect := nouveaux
		end

	mr_collision (sprite: SPRITE): BOOLEAN
			-- gestionnaire de collision entre 2 sprites. Retourne une valeur boolean true si il y a collision
		do
			if (x > sprite.x + 25.to_real) and (x < sprite.x + sprite.w_rect.to_real - 15.to_real) and (y > sprite.y + 10.to_real) and (y < sprite.y + sprite.h_rect.to_real - 20.to_real) then
				Result := True
			end
			if ((x + w) > sprite.x + 25.to_real) and ((x + w) < (sprite.x + sprite.w_rect.to_real - 15.to_real)) and (y > sprite.y + 10.to_real) and (y < sprite.y + sprite.h_rect.to_real - 20.to_real) then
				Result := True
			end
			if ((x + w) > sprite.x + 25.to_real) and ((x + w) < (sprite.x + sprite.w_rect.to_real - 15.to_real)) and ((y + h) > sprite.y + 10.to_real) and ((y + h) < sprite.y + sprite.h_rect.to_real - 20.to_real) then
				Result := True
			end
			if (x > sprite.x + 25.to_real) and (x < sprite.x + sprite.w_rect.to_real - 15.to_real) and ((y + h) > sprite.y + 10.to_real) and ((y + h) < sprite.y + sprite.h_rect.to_real - 20.to_real) then
				Result := True
			end
		end

	x: REAL_32
			-- X d'un sprite

	y: REAL_32
			-- Y d'un sprite

	w: REAL_32
			-- Largeur d'un sprite

	h: REAL_32
			-- Hauteur d'un sprite

	ptr: POINTER
			-- pointeur vers l'image d'un sprite

	x_rect: INTEGER_32 assign set_x_rect
			-- Coordonnée X du rectangle à afficher

	y_rect: INTEGER_32
			-- Coordonnée X du rectangle à afficher

	w_rect: INTEGER_32
			-- Largeur du rectangle à afficher

	h_rect: INTEGER_32
			-- Hauteur du rectangle à afficher

	text_size: INTEGER_32
			-- taille du texte

	ttf_font: POINTER
			-- pointeur TTF_Font

	texte: STRING_8
			-- String contenant le texte d'un sprite text à afficher

	sdl_color: POINTER
			-- pointeur vers la structure SDL_Color
